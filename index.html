3. Even or Odd

// evenodd.c
#include 

int main() {
    int n;
    printf("Enter an integer: ");
    scanf("%d", &n);
    if (n % 2 == 0)
        printf("%d is even\n", n);
    else
        printf("%d is odd\n", n);
    return 0;
}

4. Fibonacci Series

// fib.c
#include 

int main() {
    int n, i;
    long a = 0, b = 1, next;
    printf("Enter number of terms: ");
    scanf("%d", &n);

    for (i = 0; i < n; i++) {
        printf("%ld ", a);
        next = a + b;
        a = b;
        b = next;
    }
    return 0;
}
